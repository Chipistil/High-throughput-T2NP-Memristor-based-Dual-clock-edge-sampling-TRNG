import os
import numpy as np
import pandas as pd
import logging

logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def find_vth_and_vhold(voltage, current):
    vth = None
    vhold = None
    
    try:
        if len(voltage) != len(current):
            logging.warning("Array length mismatch")
            return None, None
        
        if len(current) == 0:
            logging.warning("Current array is empty")
            return None, None
        
        threshold = 5e-4
        vth_found = False
        
        for i, curr in enumerate(current):
            if not np.isfinite(curr):
                continue
                
            if abs(curr) > threshold:
                vth = voltage[i]
                vth_found = True
                logging.debug(f"Vth found: {vth} at index {i}")
                break
        
        if vth_found:
            for j in range(i, len(current)):
                if (j > 0 and 
                    np.isfinite(current[j-1]) and current[j-1] != 0 and 
                    np.isfinite(current[j]) and current[j] != 0):
                    try:
                        prev_current_abs = abs(current[j-1])
                        curr_current_abs = abs(current[j])
                        
                        prev_order = int(np.floor(np.log10(prev_current_abs)))
                        curr_order = int(np.floor(np.log10(curr_current_abs)))
                        
                        if (prev_order == -4 and curr_order == -5) or (prev_order == -4 and curr_order == -6):
                            vhold = voltage[j]
                            logging.debug(f"Vhold found: {vhold} at index {j}")
                            break
                        
                        if (1e-4 * 0.9 <= prev_current_abs < 1e-4 * 1.1):
                            if (1e-5 * 0.9 <= curr_current_abs < 1e-5 * 1.1) or \
                               (1e-6 * 0.9 <= curr_current_abs < 1e-6 * 1.1):
                                vhold = voltage[j]
                                logging.debug(f"Vhold (boundary case): {vhold} at index {j}")
                                break
                    except (ValueError, OverflowError) as e:
                        logging.debug(f"Error processing current data: {e}")
                        continue
    except Exception as e:
        logging.error(f"Error in find_vth_and_vhold: {e}")
    
    return vth, vhold

def main():
    input_file = r"G:\Users\Chipistil\Desktop\plotstudio\Fig.1_Pt_NbOx_Ti_IV测试\D2D_Vth与Vh箱式图\new11\09\09.xlsx"
    output_file = r"G:\Users\Chipistil\Desktop\plotstudio\Fig.1_Pt_NbOx_Ti_IV测试\D2D_Vth与Vh箱式图\new11\09\V.txt"
    
    logging.info(f"Analyzing file: {input_file}")
    
    if not os.path.exists(input_file):
        logging.error(f"File not found: {input_file}")
        return
    
    if os.path.getsize(input_file) == 0:
        logging.error(f"File is empty: {input_file}")
        return
    
    try:
        try:
            logging.info("Reading xlsx file...")
            df = pd.read_excel(input_file)
            data = df.to_numpy()
            logging.info(f"File read successfully. Shape: {data.shape}")
        except Exception as e:
            logging.error(f"Error reading xlsx: {e}")
            return
        
        if len(data.shape) == 1:
            logging.error("Insufficient data dimensions")
            return
        
        num_cols = data.shape[1]
        if num_cols < 51:
            logging.warning(f"Columns less than 51. Actual: {num_cols}")
        
        voltage = data[:, 0]
        
        if not np.all(np.isfinite(voltage)):
            logging.warning("Voltage column contains invalid values")
        
        results = []
        total_columns = min(51, num_cols)
        logging.info(f"Processing {total_columns - 1} current columns")
        
        for col in range(1, total_columns):
            current = data[:, col]
            
            if not np.any(np.isfinite(current)):
                logging.warning(f"Column {col} contains no valid data")
                results.append((None, None))
                continue
            
            vth, vhold = find_vth_and_vhold(voltage, current)
            results.append((vth, vhold))
            
            if (col) % 10 == 0 or col == total_columns - 1:
                logging.info(f"Processed {col} columns")
        
        output_dir = os.path.dirname(output_file)
        if not os.path.exists(output_dir):
            try:
                os.makedirs(output_dir)
                print(f"Created directory: {output_dir}")
            except Exception as e:
                print(f"Failed to create directory: {e}")
                return
        
        valid_results_count = 0
        try:
            with open(output_file, 'w') as f:
                f.write("Vth\tVhold\n")
                
                for vth, vhold in results:
                    vth_str = "None" if vth is None else f"{vth:.6f}"
                    vhold_str = "None" if vhold is None else f"{vhold:.6f}"
                    f.write(f"{vth_str}\t{vhold_str}\n")
                    
                    if vth is not None and vhold is not None:
                        valid_results_count += 1
            
            print(f"Analysis complete. Saved to: {output_file}")
            print(f"Total columns processed: {len(results)}")
            print(f"Valid Vth/Vhold pairs: {valid_results_count}")
            
            vth_missing = sum(1 for vth, vhold in results if vth is None)
            vhold_missing = sum(1 for vth, vhold in results if vth is not None and vhold is None)
            
            if vth_missing > 0:
                print(f"Warning: {vth_missing} columns missing Vth")
            if vhold_missing > 0:
                print(f"Warning: {vhold_missing} columns missing Vhold")
                
        except Exception as e:
            print(f"Error saving file: {e}")
            return
        
    except Exception as e:
        print(f"Runtime error: {e}")

if __name__ == "__main__":
    main()
